CREATE TABLE SEDE (
    IDSEDE INT NOT NULL PRIMARY KEY,
    CUIDAD CHARACTER(15) NOT NULL,
    DOMICILIO CHARACTER(50) NOT NULL,
    RUN_DIREC INT NOT NULL
  );  

CREATE TABLE SOCIO (
    RUN_SOCIO INT NOT NULL PRIMARY KEY,
    ID_SEDE INT NOT NULL,
    NOMSOC CHAR(15) NOT NULL,
    APELLSOC1 CHAR(15) NOT NULL,
    APELLSOC2 CHAR(15) NOT NULL,
    DIRECCIONSOC CHAR(50) NOT NULL,
    TELEFONOSOC INT NOT NULL,
    CUENTA BIGINT NOT NULL,
    BANCO CHAR(20) NOT NULL,
    FECHA_PAGO DATE NOT NULL,
    TIPO_CUOTA INT
  );

CREATE TABLE VOLUNTARIO (
    RUN_VOL INT NOT NULL PRIMARY KEY,
    TIPO_VOL INT NOT NULL ,
    NOMVOL CHAR(15) NOT NULL,
    APELLVOL1 CHAR(15) NOT NULL,
    APELLVOL2 CHAR(15) NOT NULL,
    DIRECCIONVOL CHAR(50) NOT NULL,
    TELEFVOL INT NOT NULL,
    PROFESION CHAR(30) NOT NULL,
    DISPON BOOLEAN NOT NULL,
    NUMTRABAJOS INT NOT NULL,
    CODSEDE INT NOT NULL
  );

CREATE TABLE ENVIO (
    ID_ENVIO BIGINT NOT NULL PRIMARY KEY,
    DESTINO CHARACTER(50) NOT NULL,
    FECHSALIDA DATE NOT NULL
);
CREATE TABLE DETALLEENVIO (
	COD_DETALLE BIGINT NOT NULL,
	COD_ENVIO BIGINT NOT NULL,
	UNIDAD CHARACTER(8) NOT NULL,
	ID_SEDE BIGINT NOT NULL,
	CANT BIGINT NOT NULL,
	CODMAT BIGINT NOT NULL,
    primary key (COD_DETALLE, COD_ENVIO)
);
CREATE TABLE MATERIAL (
    COD_MAT BIGINT NOT NULL PRIMARY KEY,
    TIPO CHARACTER(1) NOT NULL,
    NOMBRE CHARACTER(15) NOT null
)
